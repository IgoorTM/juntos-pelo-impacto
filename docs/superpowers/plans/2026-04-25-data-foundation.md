# Fase 0 — Data Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create the Prisma schema with all entities, enums, indexes, and onDelete policies; run the initial migration with a partial unique index via raw SQL; write the seed script for coordinators and AppConfig; implement the `getCurrentSemester` utility.

**Architecture:** Prisma schema defines six entities (User, Project, Team, TeamMember, Osc, AppConfig) with three enums (UserRole, OscStatus, ProjectStatus). The migration is generated by Prisma then manually edited to add the partial unique index on `Project.oscId`. Seed creates initial coordinators from env vars and initializes AppConfig singleton. A pure utility function `getCurrentSemester` lives in `src/common/`.

**Tech Stack:** Prisma 6, PostgreSQL 16, NestJS, TypeScript, ts-node

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

enum UserRole {
  COORDINATOR
  STUDENT
}

enum OscStatus {
  AVAILABLE
  IN_PROGRESS
  BLOCKED
}

enum ProjectStatus {
  IN_PROGRESS
  COMPLETED
  ABANDONED
  INCOMPLETE
  ONGOING
}

model User {
  id           String       @id @default(cuid())
  name         String
  email        String       @unique
  passwordHash String
  role         UserRole
  createdAt    DateTime     @default(now())
  updatedAt    DateTime     @updatedAt

  createdTeams Team[]       @relation("TeamCreator")
  teamMembers  TeamMember[]
}

model Project {
  id        String        @id @default(cuid())
  name      String        @unique
  oscId     String
  status    ProjectStatus @default(IN_PROGRESS)
  createdAt DateTime      @default(now())
  updatedAt DateTime      @updatedAt

  osc   Osc    @relation(fields: [oscId], references: [id], onDelete: Restrict)
  teams Team[]

  @@index([status])
}

model Team {
  id        String   @id @default(cuid())
  projectId String
  semester  String
  code      String   @unique
  createdBy String
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  project Project      @relation(fields: [projectId], references: [id], onDelete: Restrict)
  creator User         @relation("TeamCreator", fields: [createdBy], references: [id], onDelete: Restrict)
  members TeamMember[]

  @@index([projectId, createdAt(sort: Desc)])
}

model TeamMember {
  teamId   String
  userId   String
  joinedAt DateTime @default(now())

  team Team @relation(fields: [teamId], references: [id], onDelete: Cascade)
  user User @relation(fields: [userId], references: [id], onDelete: Restrict)

  @@id([teamId, userId])
  @@index([userId])
}

model Osc {
  id          String    @id @default(cuid())
  name        String    @unique
  description String
  email       String?
  phone       String?
  status      OscStatus @default(AVAILABLE)
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  projects Project[]

  @@index([status])
}

model AppConfig {
  id            Int      @id @default(1)
  signUpEnabled Boolean  @default(false)
  updatedAt     DateTime @updatedAt
}
```

### Task 3: Write the seed script

**Files:**
- Create: `apps/backend/prisma/seed.ts`
- Modify: `apps/backend/package.json` (add prisma.seed config)

The seed must:
1. Create coordinator users from env var `SEED_COORDINATORS` (JSON array) or a hardcoded default list
2. Initialize `AppConfig` singleton with `signUpEnabled = false`
3. Use `upsert` so it's idempotent (safe to run multiple times)

- [ ] **Step 1: Write the seed file**

Create `apps/backend/prisma/seed.ts`:

```typescript
import { PrismaClient, UserRole } from '@prisma/client';
import { hash } from 'bcrypt';

const prisma = new PrismaClient();

interface CoordinatorSeed {
  name: string;
  email: string;
  password: string;
}

const DEFAULT_COORDINATORS: CoordinatorSeed[] = [
  {
    name: 'Coordenador Admin',
    email: 'coordenador@juntos.com',
    password: 'mudar123',
  },
];

function getCoordinators(): CoordinatorSeed[] {
  const envValue = process.env.SEED_COORDINATORS;
  if (envValue) {
    return JSON.parse(envValue) as CoordinatorSeed[];
  }
  return DEFAULT_COORDINATORS;
}

async function main() {
  const coordinators = getCoordinators();

  for (const coordinator of coordinators) {
    const passwordHash = await hash(coordinator.password, 10);
    await prisma.user.upsert({
      where: { email: coordinator.email },
      update: { name: coordinator.name, passwordHash, role: UserRole.COORDINATOR },
      create: {
        name: coordinator.name,
        email: coordinator.email,
        passwordHash,
        role: UserRole.COORDINATOR,
      },
    });
    console.log(`Upserted coordinator: ${coordinator.email}`);
  }

  await prisma.appConfig.upsert({
    where: { id: 1 },
    update: {},
    create: { id: 1, signUpEnabled: false },
  });
  console.log('AppConfig initialized (signUpEnabled = false)');
}

main()
  .catch((e) => {
    console.error(e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
```

describe('getCurrentSemester', () => {
  it('returns YYYY-1 for January', () => {
    expect(getCurrentSemester(new Date('2025-01-15'))).toBe('2025-1');
  });

  it('returns YYYY-1 for June', () => {
    expect(getCurrentSemester(new Date('2025-06-30'))).toBe('2025-1');
  });

  it('returns YYYY-2 for July', () => {
    expect(getCurrentSemester(new Date('2025-07-01'))).toBe('2025-2');
  });

  it('returns YYYY-2 for December', () => {
    expect(getCurrentSemester(new Date('2025-12-31'))).toBe('2025-2');
  });

  it('handles different years', () => {
    expect(getCurrentSemester(new Date('2026-03-10'))).toBe('2026-1');
    expect(getCurrentSemester(new Date('2024-09-22'))).toBe('2024-2');
  });

  it('uses current date when no argument is provided', () => {
    const now = new Date();
    const year = now.getFullYear();
